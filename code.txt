#!binbash
 
while getopts cwe opt; do
  case $opt in
    c)
      echo -c was triggered, value $OPTARG &2
      ;;
    w)
      echo -w was triggered, value $OPTARG &2
      ;;
    e)
      echo -e was triggered, email $OPTARG &2
      ;;
    )
      echo Required Parameter -$OPTARG &2
      exit 1
      ;;
    )
      echo Option -$OPTARG requires an argument. &2
      exit 1
      ;;
  esac
done
